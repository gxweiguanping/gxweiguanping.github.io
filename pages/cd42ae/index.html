# RabbitTemplate

提示

RabbitTemplate在Spring中扮演着重要的角色，提供了一种便捷的方式来处理与RabbitMQ的交互。通过RabbitTemplate，我们可以轻松地创建、编辑、发送消息，以及在消息发送前后执行监听操作。这种高级抽象层使得操作消息更加简便，类似于原生RabbitMQ API在Spring应用中的操作方式。

具体而言，RabbitTemplate用于：

1、创建、编辑和发送消息：RabbitTemplate允许我们创建消息，设置消息内容、头部和属性等信息，并将消息发送到RabbitMQ服务器中的特定交换机和队列。

2、消息监听功能：RabbitTemplate支持在发送消息之前和发送消息之后执行监听操作。这使得我们能够在消息发送过程中进行一些自定义的处理，如记录日志、处理错误等。

3、简化操作：RabbitTemplate通过提供高级抽象，简化了与RabbitMQ的交互过程。它封装了底层的连接、通道、交换机和队列操作，使开发者能够更专注于业务逻辑，而无需过多关注底层细节。

4、整合Spring应用：RabbitTemplate与Spring框架紧密集成，可轻松地与Spring应用的其他组件（如依赖注入、AOP等）进行整合，使消息处理和应用逻辑紧密耦合。

总之，RabbitTemplate在Spring应用中起到了桥梁的作用，连接应用程序和RabbitMQ服务器，提供了高度抽象和便捷的方式来处理消息的制造和监听，使得消息通信更加方便和可控。

# 开发生产者

package com.example.rabbitmq.demo1;

import org.junit.jupiter.api.Test;
import org.springframework.amqp.rabbit.core.RabbitTemplate;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;

/**
 * @author: MrWei
 * @create: 2023-08-17 18:39
 * @Description:
 */
@SpringBootTest
public class ProviderApplicationTests {


    @Autowired
    private RabbitTemplate rabbitTemplate;

    /**
     * 生产端没有指定交换机只有routingKey和Object。
     * 消费方产生hello队列，放在默认的交换机(AMQP default)上。
     * 而默认的交换机有一个特点，只要你的routerKey的名字与这个交换机的队列有相同的名字，他就会自动路由上。
     * 生产端routingKey叫hello ，消费端生产hello队列。他们就路由上了
     */
    @Test
    void contextLoads() {
        rabbitTemplate.convertAndSend("hello", "hello world");
    }
}

package cn.publishsub1;

import org.springframework.amqp.rabbit.annotation.Exchange;
import org.springframework.amqp.rabbit.annotation.Queue;
import org.springframework.amqp.rabbit.annotation.QueueBinding;
import org.springframework.amqp.rabbit.annotation.RabbitListener;
import org.springframework.stereotype.Component;

/**
 * @Description
 * @Author MrWei
 * @Date 2023/8/17 20:44
 **/
@Component
public class PublishSubCustomer {
    @RabbitListener(bindings = @QueueBinding(
            value = @Queue, // 创建临时队列
            exchange = @Exchange(name = "logs", type = "fanout")
    ))
    public void receive1(String message) {
        System.out.println("message1 = " + message);
    }

    @RabbitListener(bindings = @QueueBinding(
            value = @Queue, //创建临时队列
            exchange = @Exchange(name = "logs", type = "fanout")  //绑定交换机类型
    ))
    public void receive2(String message) {
        System.out.println("message2 = " + message);
    }
}

package cn.direct1;

import org.springframework.amqp.rabbit.annotation.Exchange;
import org.springframework.amqp.rabbit.annotation.Queue;
import org.springframework.amqp.rabbit.annotation.QueueBinding;
import org.springframework.amqp.rabbit.annotation.RabbitListener;
import org.springframework.stereotype.Component;

/**
 * @Description
 * @Author MrWei
 * @Date 2023/8/17 20:47
 **/
@Component
public class DirectCustomer {
    @RabbitListener(bindings = {
            @QueueBinding(
                    value = @Queue, // 创建临时队列
                    key = {"info", "error"}, // 路由key
                    exchange = @Exchange(type = "direct", name = "directs")
            )})
    public void receive1(String message) {
        System.out.println("message1 = " + message);
    }

    @RabbitListener(bindings = {
            @QueueBinding(
                    value = @Queue,
                    key = {"error"},
                    exchange = @Exchange(type = "direct", name = "directs")
            )})
    public void receive2(String message) {
        System.out.println("message2 = " + message);
    }
}

package cn.topic1;

import org.springframework.amqp.rabbit.annotation.Exchange;
import org.springframework.amqp.rabbit.annotation.Queue;
import org.springframework.amqp.rabbit.annotation.QueueBinding;
import org.springframework.amqp.rabbit.annotation.RabbitListener;
import org.springframework.stereotype.Component;

/**
 * @Description
 * @Author MrWei
 * @Date 2023/8/17 20:50
 **/
@Component
public class TopicCustomer {
    @RabbitListener(bindings = {
            @QueueBinding(
                    value = @Queue,
                    key = {"user.*"},
                    exchange = @Exchange(type = "topic",name = "topics")
            )
    })
    public void receive1(String message){
        System.out.println("message1 = " + message);
    }

    @RabbitListener(bindings = {
            @QueueBinding(
                    value = @Queue,
                    key = {"user.#"},
                    exchange = @Exchange(type = "topic",name = "topics")
            )
    })
    public void receive2(String message){
        System.out.println("message2 = " + message);
    }
}
