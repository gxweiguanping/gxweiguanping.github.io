import lombok.AllArgsConstructor;
import lombok.Data;
import lombok.NoArgsConstructor;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;
import org.springframework.validation.annotation.Validated;

@Data
@Validated
@NoArgsConstructor
@AllArgsConstructor
@Component
public class InfluxProperties {
    /**
     * InfluxDB服务器的URL。该字段必须为非空字符串。
     */
    @Value("${spring.influx.url:''}")
    private String url;

    /**
     * 用户名，用于身份验证。可选字段，可以为空。
     */
    @Value("${spring.influx.user:''}")
    private String username;


    /**
     * 密码，用于身份验证。可选字段，可以为空。
     */
    @Value("${spring.influx.password:''}")
    private String password;

    /**
     * 需要连接的数据库
     */
    @Value("${spring.influx.database}")
    public String database;

    /**
     * 保留策略
     */
    @Value("${spring.influx.retention-policy:''}")
    public String retentionPolicy;
}

import com.aiurt.influxdb.util.InfluxDbUtils;

# 创建InfluxDB工具类

import lombok.extern.slf4j.Slf4j;
import org.influxdb.InfluxDB;
import org.influxdb.InfluxDBFactory;
import org.influxdb.dto.Query;

@Slf4j
public class InfluxDbUtils {

    /**
     * InfluxDB服务器的用户名。
     */
    private String username;

    /**
     * InfluxDB服务器的密码。
     */
    private String password;

    /**
     * InfluxDB服务器的URL。
     */
    private String url;

    /**
     * InfluxDB服务器的URL。
     */
    public String database;

    /**
     * 要使用的保留策略（retention policy）名称。
     */
    private String retentionPolicy;

    /**
     * InfluxDB的客户端对象，用于与InfluxDB服务器进行交互。
     */
    private InfluxDB influxDB;

    /**
     * 构造一个InfluxDbUtils对象，用于与InfluxDB服务器建立连接和交互。
     *
     * @param username         InfluxDB服务器的用户名。
     * @param password         InfluxDB服务器的密码。
     * @param url              InfluxDB服务器的URL。
     * @param database         要连接的InfluxDB数据库名称。
     * @param retentionPolicy  要使用的保留策略（retention policy）名称。
     */
    public InfluxDbUtils(String username, String password, String url, String database,
                         String retentionPolicy) {
        this.username = username;
        this.password = password;
        this.url = url;
        this.database = database;
        this.retentionPolicy = retentionPolicy == null || "".equals(retentionPolicy) ? "autogen" : retentionPolicy;
        this.influxDB = influxDbBuild();
    }

    /**
     * 连接数据库 ，若不存在则创建
     *
     * @return influxDb实例
     */
    private InfluxDB influxDbBuild() {
        if (influxDB == null) {
            influxDB = InfluxDBFactory.connect(url, username, password);
        }
        try {
            createDB(database);
            influxDB.setDatabase(database);
        } catch (Exception e) {
            log.error("create influx db failed, error: {}", e.getMessage());
        } finally {
            influxDB.setRetentionPolicy(retentionPolicy);
        }
        influxDB.setLogLevel(InfluxDB.LogLevel.BASIC);
        return influxDB;
    }

    /****
     *  创建数据库
     * @param database 数据库名称
     */
    private void createDB(String database) {
        influxDB.query(new Query("CREATE DATABASE " + database));
    }

    public String getDatabase() {
        return database;
    }

    public InfluxDB getInfluxDB() {
        return influxDB;
    }
}

# 创建InfluxDB插入DTO

import lombok.Data;
import java.util.Map;

@Data
public class InfluxDbDataPointDTO {
    /**
     * 插入的测量名称
     */
    private String measurement;

    /**
     * 数据点的时间戳。通常是一个长整型或日期时间格式。用于指定数据点的时间。
     */
    private Long time;

    /**
     * 标签(tags)是用于标识和过滤数据点的关键信息。它是一个键值对的映射，其中键是标签名，值是标签值。
     * 标签通常用于筛选查询或组织数据。
     */
    private Map<String, String> tags;

    /**
     * 字段(fields)包含实际的数据。它也是一个键值对的映射，其中键是字段名，值是字段的值。
     * 字段存储实际的数据内容。
     */
    private Map<String, Object> fields;
}

# 创建InfluxDB服务接口

import com.aiurt.influxdb.dto.InfluxDbDataPointDTO;
import org.influxdb.InfluxDB;
import org.influxdb.dto.QueryResult;
import java.util.List;

public interface InfluxService {
    /**
     * 测试与InfluxDB的连接是否正常。
     *
     * @return 如果连接正常，返回true；否则返回false。
     */
    Boolean ping();

    /**
     * 创建一个或多个数据库。
     *
     * @param dataBaseName 要创建的数据库名称（可变参数，可以一次创建多个数据库）。
     */
    void createDataBase(String... dataBaseName);

    /**
     * 删除一个或多个数据库。
     *
     * @param dataBaseName 要删除的数据库名称（可变参数，可以一次删除多个数据库）。
     */
    void deleteDataBase(String... dataBaseName);

    /**
     * 批量写入数据
     *
     * @param consistency 一致性
     * @param records     要保存的数据
     */
    void batchInsert(final InfluxDB.ConsistencyLevel consistency, final List<InfluxDbDataPointDTO> records);

    /**
     * 向InfluxDB插入数据点（Point）。
     *
     * @param influxDbDataPointDto 包含要插入的数据点信息的数据传输对象。
     */
    void insert(InfluxDbDataPointDTO influxDbDataPointDto);

    /**
     * 执行InfluxDB查询并返回查询结果。
     *
     * @param command InfluxDB查询命令。
     * @return 查询结果，通常是一个包含数据点的结果集。
     */
    QueryResult query(String command);

    /**
     * 处理InfluxDB查询结果并将其映射为指定类型的对象列表。
     *
     * @param command     InfluxDB查询命令。
     * @param resultClass 目标对象的类，用于映射查询结果。
     * @param <T>         目标对象的泛型类型。
     * @return 包含查询结果映射的对象列表。
     */
    <T> List<T> query(String command, Class<T> resultClass);

    /**
     * 处理InfluxDB查询结果，以计算结果数据的数量或其他统计信息。
     *
     * @param queryResult 查询InfluxDB后获得的查询结果。
     * @return 查询结果中的数据点数量或其他统计信息。
     */
    long countResultProcess(QueryResult queryResult);

}

import cn.hutool.core.collection.CollUtil;
import cn.hutool.core.map.MapUtil;
import com.aiurt.influxdb.dto.InfluxDbDataPointDTO;
import com.aiurt.influxdb.service.InfluxService;
import com.aiurt.influxdb.util.InfluxDbUtils;
import lombok.extern.slf4j.Slf4j;
import org.influxdb.InfluxDB;
import org.influxdb.dto.*;
import org.springframework.beans.BeanWrapperImpl;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.sql.Timestamp;
import java.time.Instant;
import java.time.ZonedDateTime;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.concurrent.TimeUnit;

/**
 * @create: 2023-10-25 16:50
 * @Description: Influx服务接口实现类
 */
@Slf4j
@Service
public class InfluxServiceImpl implements InfluxService {
    @Autowired
    private InfluxDbUtils influxDbUtils;

    @Override
    public Boolean ping() {
        boolean isConnected = false;
        InfluxDB influxDB = influxDbUtils.getInfluxDB();
        Pong pong;
        try {
            pong = influxDB.ping();
            if (pong != null) {
                isConnected = true;
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
        return isConnected;
    }

    @Override
    public void createDataBase(String... dataBaseName) {
        InfluxDB influxDB = influxDbUtils.getInfluxDB();
        if (dataBaseName.length > 0) {
            // CREATE DATABASE 表示纯inclufx cmd命令
            influxDB.query(new Query("CREATE DATABASE " + dataBaseName[0]));
            return;
        }
        if (influxDbUtils.getDatabase() == null) {
            log.error("如参数不指定数据库名,配置文件 spring.influx.dataBaseName 必须指定");
            return;
        }
        influxDB.query(new Query("CREATE DATABASE " + influxDbUtils.getDatabase()));
    }

    @Override
    public void deleteDataBase(String... dataBaseName) {
        InfluxDB influxDB = influxDbUtils.getInfluxDB();
        if (dataBaseName.length > 0) {
            // CREATE DATABASE 表示纯inclufx cmd命令
            influxDB.query(new Query("DROP DATABASE " + dataBaseName[0]));
            return;
        }
        if (influxDbUtils.getDatabase() == null) {
            log.error("如参数不指定数据库名,配置文件 spring.influx.dataBaseName 必须指定");
            return;
        }
        influxDB.query(new Query("DROP DATABASE " + influxDbUtils.getDatabase()));
    }

    @Override
    public void batchInsert(InfluxDB.ConsistencyLevel consistency, List<InfluxDbDataPointDTO> records) {
        if (CollUtil.isEmpty(records)) {
            return;
        }

        InfluxDB influxDB = influxDbUtils.getInfluxDB();
        String databaseName = influxDbUtils.getDatabase();

        BatchPoints batchPoints = BatchPoints
                .database(databaseName)
                .retentionPolicy("")
                .consistency(consistency)
                .build();

        for (InfluxDbDataPointDTO record : records) {
            Point.Builder builder = Point.measurement(record.getMeasurement())
                    .time(record.getTime(), java.util.concurrent.TimeUnit.MILLISECONDS);

            if (MapUtil.isNotEmpty(record.getTags())) {
                builder.tag(record.getTags());
            }

            if (MapUtil.isNotEmpty(record.getFields())) {
                builder.fields(record.getFields());
            }

            Point point = builder.build();
            batchPoints.point(point);
        }

        // 批量写入数据
        influxDB.write(batchPoints);
    }

    @Override
    public void insert(InfluxDbDataPointDTO influxDbDataPointDto) {
        InfluxDB influxDB = influxDbUtils.getInfluxDB();
        String databaseName = influxDbUtils.getDatabase();
        Point.Builder builder = Point.measurement(influxDbDataPointDto.getMeasurement());
        if (MapUtil.isNotEmpty(influxDbDataPointDto.getTags())) {
            builder.tag(influxDbDataPointDto.getTags());
        }

        if (MapUtil.isNotEmpty(influxDbDataPointDto.getFields())) {
            builder.fields(influxDbDataPointDto.getFields());
        }

        if (influxDbDataPointDto.getTime() != null) {
            builder.time(influxDbDataPointDto.getTime(), TimeUnit.MILLISECONDS);
        }

        influxDB.write(databaseName, "", builder.build());
    }

    @Override
    public QueryResult query(String command) {
        InfluxDB influxDB = influxDbUtils.getInfluxDB();
        String databaseName = influxDbUtils.getDatabase();
        Query query = new Query(command, databaseName);
        return influxDB.query(query);
    }

    @Override
    public <T> List<T> query(String command, Class<T> resultClass) {
        QueryResult queryResult = query(command);
        return queryResultProcess(queryResult, resultClass);
    }

    @Override
    public long countResultProcess(QueryResult queryResult) {
        long count = 0;
        List<Map<String, Object>> list = queryResultProcess(queryResult);
        if (list != null && list.size() != 0) {
            Map<String, Object> map = list.get(0);
            double num = (Double) map.get("count");
            count = new Double(num).longValue();
        }
        return count;
    }

    /**
     * @desc 查询结果处理
     */
    public List<Map<String, Object>> queryResultProcess(QueryResult queryResult) {
        List<Map<String, Object>> mapList = new ArrayList<>();
        List<QueryResult.Result> resultList = queryResult.getResults();

        // 把查询出的结果集转换成对应的实体对象，聚合成list
        for (QueryResult.Result query : resultList) {
            List<QueryResult.Series> seriesList = query.getSeries();
            if (seriesList != null && seriesList.size() != 0) {
                for (QueryResult.Series series : seriesList) {
                    List<String> columns = series.getColumns();
                    String[] keys = columns.toArray(new String[columns.size()]);
                    List<List<Object>> values = series.getValues();
                    if (values != null && values.size() != 0) {
                        for (List<Object> value : values) {
                            Map<String, Object> map = new HashMap(keys.length);
                            for (int i = 0; i < keys.length; i++) {
                                map.put(keys[i], value.get(i));
                            }
                            mapList.add(map);
                        }
                    }
                }
            }
        }
        return mapList;
    }

    /**
     * 处理InfluxDB查询结果并将其映射为指定类型的对象列表。
     *
     * @param queryResult InfluxDB查询的结果对象。
     * @param resultClass 目标对象的类，用于映射查询结果。
     * @param <T>         目标对象的泛型类型。
     * @return 包含查询结果映射的对象列表。
     */
    public <T> List<T> queryResultProcess(QueryResult queryResult, Class<T> resultClass) {
        List<T> resultList = new ArrayList<>();
        // 1.获取结果
        List<QueryResult.Result> queryResults = queryResult.getResults();

        // 2.遍历结果
        for (QueryResult.Result result : queryResults) {
            // 3.获取series
            List<QueryResult.Series> seriesList = result.getSeries();

            if (seriesList != null && !seriesList.isEmpty()) {
                // 4.遍历series
                for (QueryResult.Series series : seriesList) {
                    // 5.获取所有列
                    List<String> columns = series.getColumns();
                    // 6.获取所有值
                    List<List<Object>> values = series.getValues();
                    // 7.获取所有的列名
                    String[] keys = columns.toArray(new String[columns.size()]);

                    if (values != null && !values.isEmpty()) {
                        // 8.遍历数据，获取结构
                        for (List<Object> value : values) {
                            // 9.根据resultClass进行封装
                            T instance = createInstance(resultClass);
                            // 10.通过Spring框架提供反射类进行处理
                            BeanWrapperImpl beanWrapper = new BeanWrapperImpl(instance);

                            Map<String, Object> fields = new HashMap<>(16);
                            for (int i = 0; i < keys.length; i++) {

                                String val = String.valueOf(value.get(i));

                                if ("time".equals(keys[i])) {
                                    ZonedDateTime parse = ZonedDateTime.parse(val);
                                    Instant instant = parse.toInstant();
                                    long time = Timestamp.from(instant).getTime();
                                    beanWrapper.setPropertyValue("time", time);
                                } else {
                                    fields.put(keys[i], val);
                                }

                            }
                            beanWrapper.setPropertyValue("fields", fields);
                            resultList.add(instance);
                        }
                    }
                }
            }
        }

        return resultList;
    }

    /**
     * 创建泛型对象实例
     *
     * @param clazz
     * @param <T>
     * @return
     */
    private <T> T createInstance(Class<T> clazz) {
        T instance = null;
        try {
            instance = clazz.newInstance();
        } catch (InstantiationException | IllegalAccessException e) {
            log.error(e.getMessage(), e);
        }
        return instance;
    }

}
