    @Override
    public void batchInsert(InfluxDB.ConsistencyLevel consistency, List<InfluxDbDataPointDTO> records) {
        if (CollUtil.isEmpty(records)) {
            return;
        }

        InfluxDB influxDB = influxDbUtils.getInfluxDB();
        String databaseName = influxDbUtils.getDatabase();

        BatchPoints batchPoints = BatchPoints
                .database(databaseName)
                .retentionPolicy("")
                .consistency(consistency)
                .build();

        for (InfluxDbDataPointDTO record : records) {
            Point.Builder builder = Point.measurement(record.getMeasurement())
                    .time(record.getTime(), java.util.concurrent.TimeUnit.MILLISECONDS);

            if (MapUtil.isNotEmpty(record.getTags())) {
                builder.tag(record.getTags());
            }

            if (MapUtil.isNotEmpty(record.getFields())) {
                builder.fields(record.getFields());
            }

            Point point = builder.build();
            batchPoints.point(point);
        }

        // æ‰¹é‡å†™å…¥æ•°æ®
        influxDB.write(batchPoints);
    }

    @Override
    public void insert(InfluxDbDataPointDTO influxDbDataPointDto) {
        InfluxDB influxDB = influxDbUtils.getInfluxDB();
        String databaseName = influxDbUtils.getDatabase();
        Point.Builder builder = Point.measurement(influxDbDataPointDto.getMeasurement());
        if (MapUtil.isNotEmpty(influxDbDataPointDto.getTags())) {
            builder.tag(influxDbDataPointDto.getTags());
        }

        if (MapUtil.isNotEmpty(influxDbDataPointDto.getFields())) {
            builder.fields(influxDbDataPointDto.getFields());
        }

        if (influxDbDataPointDto.getTime() != null) {
            builder.time(influxDbDataPointDto.getTime(), TimeUnit.MILLISECONDS);
        }

        influxDB.write(databaseName, "", builder.build());
    }

    @Override
    public QueryResult query(String command) {
        InfluxDB influxDB = influxDbUtils.getInfluxDB();
        String databaseName = influxDbUtils.getDatabase();
        Query query = new Query(command, databaseName);
        return influxDB.query(query);
    }

    @Override
    public <T> List<T> query(String command, Class<T> resultClass) {
        QueryResult queryResult = query(command);
        return queryResultProcess(queryResult, resultClass);
    }

    @Override
    public long countResultProcess(QueryResult queryResult) {
        long count = 0;
        List<Map<String, Object>> list = queryResultProcess(queryResult);
        if (list != null && list.size() != 0) {
            Map<String, Object> map = list.get(0);
            double num = (Double) map.get("count");
            count = new Double(num).longValue();
        }
        return count;
    }

    /**
     * @desc æŸ¥è¯¢ç»“æœå¤„ç†
     */
    public List<Map<String, Object>> queryResultProcess(QueryResult queryResult) {
        List<Map<String, Object>> mapList = new ArrayList<>();
        List<QueryResult.Result> resultList = queryResult.getResults();

        // æŠŠæŸ¥è¯¢å‡ºçš„ç»“æœé›†è½¬æ¢æˆå¯¹åº”çš„å®ä½“å¯¹è±¡ï¼Œèšåˆæˆlist
        for (QueryResult.Result query : resultList) {
            List<QueryResult.Series> seriesList = query.getSeries();
            if (seriesList != null && seriesList.size() != 0) {
                for (QueryResult.Series series : seriesList) {
                    List<String> columns = series.getColumns();
                    String[] keys = columns.toArray(new String[columns.size()]);
                    List<List<Object>> values = series.getValues();
                    if (values != null && values.size() != 0) {
                        for (List<Object> value : values) {
                            Map<String, Object> map = new HashMap(keys.length);
                            for (int i = 0; i < keys.length; i++) {
                                map.put(keys[i], value.get(i));
                            }
                            mapList.add(map);
                        }
                    }
                }
            }
        }
        return mapList;
    }

    public <T> List<T> queryResultProcess(QueryResult queryResult, Class<T> resultClass) {
        List<T> resultList = new ArrayList<>();
        // 1.è·å–ç»“æœ
        List<QueryResult.Result> queryResults = queryResult.getResults();

        // 2.éå†ç»“æœ
        for (QueryResult.Result result : queryResults) {
            // 3.è·å–series
            List<QueryResult.Series> seriesList = result.getSeries();

            if (seriesList != null && !seriesList.isEmpty()) {
                // 4.éå†series
                for (QueryResult.Series series : seriesList) {
                    // 5.è·å–æ‰€æœ‰åˆ—
                    List<String> columns = series.getColumns();
                    // 6.è·å–æ‰€æœ‰å€¼
                    List<List<Object>> values = series.getValues();
                    // 7.è·å–æ‰€æœ‰çš„åˆ—å
                    String[] keys = columns.toArray(new String[columns.size()]);

                    if (values != null && !values.isEmpty()) {
                        // 8.éå†æ•°æ®ï¼Œè·å–ç»“æ„
                        for (List<Object> value : values) {
                            // 9.æ ¹æ®resultClassè¿›è¡Œå°è£…
                            T instance = createInstance(resultClass);
                            // 10.é€šè¿‡Springæ¡†æ¶æä¾›åå°„ç±»è¿›è¡Œå¤„ç†
                            BeanWrapperImpl beanWrapper = new BeanWrapperImpl(instance);

                            Map<String, Object> fields = new HashMap<>(16);
                            for (int i = 0; i < keys.length; i++) {

                                String val = String.valueOf(value.get(i));

                                if ("time".equals(keys[i])) {
                                    ZonedDateTime parse = ZonedDateTime.parse(val);
                                    Instant instant = parse.toInstant();
                                    long time = Timestamp.from(instant).getTime();
                                    beanWrapper.setPropertyValue("time", time);
                                } else {
                                    fields.put(keys[i], val);
                                }

                            }
                            beanWrapper.setPropertyValue("fields", fields);
                            resultList.add(instance);
                        }
                    }
                }
            }
        }

        return resultList;
    }
