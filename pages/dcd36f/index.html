# docker-compose.yml

version: '3'
services:
  zookeeper:
    image: wurstmeister/zookeeper
    container_name: zookeeper
    hostname: zookeeper
    volumes:
       - /home/docker/docker_zookeeper/data:/opt/zookeeper-3.4.13/data
       - /home/docker/docker_zookeeper/conf:/opt/zookeeper-3.4.13/conf
       - /home/docker/docker_zookeeper/secrets:/opt/zookeeper-3.4.13/secrets/
    environment:
      ZOOKEEPER_CLIENT_PORT: 2181
      ZOOKEEPER_TICK_TIME: 2000
      SERVER_JVMFLAGS: -Djava.security.auth.login.config=/opt/zookeeper-3.4.13/secrets/server_jaas.conf
    ports:
      - 2181:2181
    restart: always
    privileged: true
  kafka1:
    image: wurstmeister/kafka
    container_name: kafka_1
    hostname: kafka_1
    depends_on:
      - zookeeper
    ports:
      - "9092:9092"
    volumes:
      - /home/docker/docker_kafka1/data:/kafka
      - /home/docker/docker_kafka1/secrets:/opt/kafka/secrets/
      - /var/run/docker.sock:/var/run/docker.sock
    environment:
      KAFKA_BROKER_ID: 1  
      KAFKA_NUM_PARTITIONS: 2
      KAFKA_DEFAULT_REPLICATION_FACTOR: 2
      KAFKA_ZOOKEEPER_CONNECT: zookeeper:2181
      KAFKA_LISTENERS: SASL_PLAINTEXT://0.0.0.0:9092
      KAFKA_ADVERTISED_LISTENERS: SASL_PLAINTEXT://{ip}:9092
      KAFKA_SECURITY_INTER_BROKER_PROTOCOL: SASL_PLAINTEXT	  
      KAFKA_SASL_MECHANISM_INTER_BROKER_PROTOCOL: PLAIN
      KAFKA_SASL_ENABLED_MECHANISMS: PLAIN
      KAFKA_AUTHORIZER_CLASS_NAME: kafka.security.auth.SimpleAclAuthorizer
      KAFKA_SUPER_USERS: User:admin
      KAFKA_ALLOW_EVERYONE_IF_NO_ACL_FOUND: "true"
      KAFKA_OPTS: -Djava.security.auth.login.config=/opt/kafka/secrets/server_jaas.conf
    restart: always
    privileged: true
  kafka2:
    image: wurstmeister/kafka
    container_name: kafka_2
    hostname: kafka_2
    depends_on:
      - zookeeper
    ports:
      - "9093:9092"
    volumes:
      - /home/docker/docker_kafka2/data:/kafka
      - /home/docker/docker_kafka2/secrets:/opt/kafka/secrets/
      - /var/run/docker.sock:/var/run/docker.sock
    environment:
      KAFKA_BROKER_ID: 2  
      KAFKA_NUM_PARTITIONS: 2
      KAFKA_DEFAULT_REPLICATION_FACTOR: 2

      KAFKA_ADVERTISED_LISTENERS: SASL_PLAINTEXT://{ip}:9093
      KAFKA_SECURITY_INTER_BROKER_PROTOCOL: SASL_PLAINTEXT	  
      KAFKA_SASL_MECHANISM_INTER_BROKER_PROTOCOL: PLAIN
      KAFKA_SASL_ENABLED_MECHANISMS: PLAIN
      KAFKA_AUTHORIZER_CLASS_NAME: kafka.security.auth.SimpleAclAuthorizer

      KAFKA_ALLOW_EVERYONE_IF_NO_ACL_FOUND: "true"
